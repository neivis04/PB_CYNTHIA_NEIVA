select distinct
    A.nome
    
from 
    AUTOR A

join 
    LIVRO L on A.codAutor = L.autor
    
join 
    EDITORA E on L.editora = E.codEditora
    
join 
    ENDERECO EN on E.endereco = EN.codEndereco
    
where 
    EN.estado not in ('PARAN√Å', 'SANTA CATARINA', 'RIO GRANDE DO SUL')
    
order by 
    A.nome asc;
