SELECT 
    v.nmvdd AS vendedor,
    SUM(ve.qtd * ve.vrunt) AS valor_total_vendas,
    ROUND(SUM(ve.qtd * ve.vrunt) * (v.perccomissao / 100), 2) AS comissao
FROM 
    tbvendedor v
JOIN 
    tbvendas ve ON v.cdvdd = ve.cdvdd
WHERE 
    ve.status = 'Conclu√≠do'
GROUP BY 
    v.nmvdd
ORDER BY 
    comissao DESC;
