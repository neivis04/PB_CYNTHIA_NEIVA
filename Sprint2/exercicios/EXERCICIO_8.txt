select
    V.cdvdd, V.nmvdd
    
from 
    TBVENDEDOR V
    
join 
    TBVENDAS VD on V.cdvdd = VD.cdvdd

where 
    VD.status = 'Concluído'
    
group by 
    V.cdvdd, V.nmvdd
    
having 
    count(*) = (
        select max(venda_count)
        from (
            select cdvdd, count(*) as venda_count
            from TBVENDAS
            where status = 'Concluído'
            group by cdvdd
        ) as vendas_por_vendedor
    );
