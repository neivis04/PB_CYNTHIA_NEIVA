
select
  L.cod,
  L.titulo,
  A.codAutor as autor,
  E.codEditora as editora,
  L.valor,
  L.publicacao,
  L.edicao,
  L.idiom
  
from
    LIVRO as L

join 
    AUTOR as A
    on L.autor = A.codAutor
join 
    EDITORA as E
    in L.editora = E.codEditora
    
where
    STRFTIME('%Y', L.publicacao) > '2014'
  
order by
    L.cod;
